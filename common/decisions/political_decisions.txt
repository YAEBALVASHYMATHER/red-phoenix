	ignite_the_fascist_civil_war = {

		}

		visible = {
			has_country_flag = preparation_for_fascist_civil_war
			OR = {
				has_idea_with_trait = fascist_demagogue
				has_idea_with_trait = syncretic_revanchist
			}

			check_has_focus_tree_to_switch_to_nationalism = yes
			check_variable = { num_owned_states > 1 }
		}
	
		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = liberalism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@liberalism
				}
			}
			if = {
				limit = {
					has_government = communist
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@communist
				}
			}
			if = {
				limit = {
					has_government = despotism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@despotism
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = nationalism
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_nationalism_civil_war
			clr_country_flag = military_support_for_nationalism_civil_war
			clr_country_flag = civil_support_for_nationalism_civil_war
			set_country_flag = ideology_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
